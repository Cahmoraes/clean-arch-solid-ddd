// apps/frontend/src/features/auth/schemas/index.ts
export const definePasswordSchema = z
	.object({
		newPassword: passwordSchema,
		confirmPassword: z.string().min(1, "Confirme sua nova senha."),
	})
	.refine((data) => data.newPassword === data.confirmPassword, {
		path: ["confirmPassword"],
		message: "A confirmação não corresponde à nova senha.",
	})

export type DefinePasswordInput = z.infer<typeof definePasswordSchema>

// apps/frontend/src/features/auth/api/index.ts
import { useQueryClient } from "@tanstack/react-query"
import { profileQueryKeys } from "@/features/profile/api"

export function useCreatePasswordReauthGrant(): UseMutationResult<
	{ reauthGrant: string; expiresInSeconds: number },
	ApiError,
	{ provider: "google"; idToken: string }
> {
	return useMutation({
		mutationFn: async (input) => {
			const { data, error } = await api.POST("/users/me/password/reauth", {
				body: input,
			})
			if (error || !data) throw toApiError(error)
			return data
		},
	})
}

export function useDefinePassword(): UseMutationResult<
	void,
	ApiError,
	{ provider: "google"; reauthGrant: string; newPassword: string }
> {
	const queryClient = useQueryClient()
	return useMutation({
		mutationFn: async (input) => {
			const { error } = await api.POST("/users/me/password", {
				body: {
					provider: input.provider,
					reauthGrant: input.reauthGrant,
					newRawPassword: input.newPassword,
				},
			})
			if (error) throw toApiError(error)
		},
		onSuccess: async () => {
			await queryClient.invalidateQueries({ queryKey: profileQueryKeys.me() })
		},
	})
}

export function useChangePassword(): UseMutationResult<
	void,
	ApiError,
	ChangePasswordInput
> {
	const queryClient = useQueryClient()
	return useMutation({
		mutationFn: async (input) => {
			const { error } = await api.PATCH("/users/me/change-password", {
				body: {
					currentRawPassword: input.currentPassword,
					newRawPassword: input.newPassword,
				},
			})
			if (error) throw toApiError(error)
		},
		onSuccess: async () => {
			await queryClient.invalidateQueries({ queryKey: profileQueryKeys.me() })
		},
	})
}

// apps/frontend/src/app/(authenticated)/perfil/senha/page.tsx
function DefinePasswordForm({ provider }: { provider: "google" }) {
	const [reauthGrant, setReauthGrant] = useState<string | null>(null)
	const { mutateAsync: requestReauth, isPending: isReauthPending } =
		useCreatePasswordReauthGrant()
	const { mutateAsync: definePassword, isPending, error } = useDefinePassword()
	const form = useForm<DefinePasswordInput>({
		resolver: zodResolver(definePasswordSchema),
		defaultValues: { newPassword: "", confirmPassword: "" },
	})

	async function handleGoogleSuccess(idToken: string) {
		const result = await requestReauth({ provider, idToken })
		setReauthGrant(result.reauthGrant)
		toast.success("Reautenticação confirmada. Agora você já pode definir sua senha.")
	}

	async function onSubmit(values: DefinePasswordInput) {
		if (!reauthGrant) {
			toast.error("Reautentique com o provider antes de definir sua senha.")
			return
		}
		await definePassword({
			provider,
			reauthGrant,
			newPassword: values.newPassword,
		})
		form.reset()
		toast.success("Senha definida com sucesso.")
	}

	return (
		<form onSubmit={form.handleSubmit(onSubmit)} className="flex flex-col gap-4">
			<FormField label="Nova senha" type="password" {...form.register("newPassword")} />
			<FormField label="Confirmar nova senha" type="password" {...form.register("confirmPassword")} />
			<GoogleSignInButton onSuccess={handleGoogleSuccess} disabled={isPending} isPending={isReauthPending} />
			{error ? <p data-testid="change-password-submit-error">{changePasswordErrorMessage(error)}</p> : null}
			<Button type="submit" disabled={isPending || !reauthGrant} data-testid="change-password-submit">
				{isPending ? "Definindo…" : "Definir senha"}
			</Button>
		</form>
	)
}

export default function ChangePasswordPage() {
	const { data, isLoading } = useMe()
	if (isLoading || !data) return null
	const externalProvider = data.authMethods.find((method) => method !== "password")
	const isDefinePasswordFlow = !data.hasPassword && externalProvider === "google"

	return (
		<section className="mx-auto flex w-full max-w-md flex-col gap-8">
			<header className="flex flex-col gap-2">
				<h1 className="font-display text-3xl font-medium tracking-tight text-foreground">
					{isDefinePasswordFlow ? "Definir senha" : "Alterar senha"}
				</h1>
				<p className="text-sm text-muted-foreground">
					{isDefinePasswordFlow
						? "Confirme sua identidade com o provider e crie uma senha local."
						: "Use uma senha forte e diferente da atual."}
				</p>
			</header>
			{isDefinePasswordFlow ? (
				<DefinePasswordForm provider="google" />
			) : (
				<ExistingChangePasswordForm />
			)}
		</section>
	)
}

// apps/frontend/src/app/(public)/login/page.tsx
function loginErrorMessage(error: unknown): string {
	if (error instanceof ApiError) {
		if (error.code === "password_not_set") {
			return "Essa conta ainda não possui senha local. Entre com o provider externo ou defina uma senha no perfil."
		}
		if (error.status === 401) return "E-mail ou senha incorretos."
		return error.userMessage
	}
	return "Não foi possível concluir o login. Tente novamente."
}
